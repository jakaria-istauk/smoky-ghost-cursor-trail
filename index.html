Usage Example

// Basic usage
const ghostTrail = new SmokyGhostTrail('#my-element', {
    size: 0.1,
    tail: {
        dotsNumber: 25,
        spring: 1.4,
        friction: 0.3,
        gravity: 0
    },
    mainColor: [0.98, 0.96, 0.96],
    borderColor: [0.2, 0.5, 0.7],
    isFlatColor: false
});

// Start the effect
ghostTrail.start();

// Update options dynamically
ghostTrail.updateOptions({
    size: 0.15,
    mainColor: [1.0, 0.5, 0.5]
});

// Stop and clean up
ghostTrail.stop();
ghostTrail.destroy();
